    59тЖТ    59тЖТfunction relativeTime(timestamp: string): string {
    60тЖТ    60тЖТ  const now = Date.now();
    61тЖТ    61тЖТ  const then = new Date(timestamp).getTime();
    62тЖТ    62тЖТ  const diff = now - then;
    63тЖТ    63тЖТ
    64тЖТ    64тЖТ  const minutes = Math.floor(diff / 60000);
    65тЖТ    65тЖТ  if (minutes < 1) return 'Ahora mismo';
    66тЖТ    66тЖТ  if (minutes < 60) return `Hace ${minutes} min`;
    67тЖТ    67тЖТ
    68тЖТ    68тЖТ  const hours = Math.floor(minutes / 60);
    69тЖТ    69тЖТ  if (hours < 24) return `Hace ${hours} hora${hours > 1 ? 's' : ''}`;
    70тЖТ    70тЖТ
    71тЖТ    71тЖТ  const days = Math.floor(hours / 24);
    72тЖТ    72тЖТ  if (days < 30) return `Hace ${days} d├нa${days > 1 ? 's' : ''}`;
    73тЖТ    73тЖТ
    74тЖТ    74тЖТ  const months = Math.floor(days / 30);
    75тЖТ    75тЖТ  return `Hace ${months} mes${months > 1 ? 'es' : ''}`;
    76тЖТ    76тЖТ}
