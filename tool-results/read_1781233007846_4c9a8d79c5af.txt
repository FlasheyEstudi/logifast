    71→  const {
    72→    activeModule, setActiveModule, moduleFade, alerts,
    73→    commandPaletteOpen, setCommandPaletteOpen,
    74→    simulationRunning, toggleSimulation,
    75→    simulateNewOrder, simulateDelivery, simulateStatusChange, updateMotoPositions,
    76→  } = useStore();
    77→
    78→  const [avatarOpen, setAvatarOpen] = useState(false);
    79→  const [fabOpen, setFabOpen] = useState(false);
    80→  const [isFullscreen, setIsFullscreen] = useState(false);
    81→  const [showHelpOverlay, setShowHelpOverlay] = useState(false);
    82→  const [moreMenuOpen, setMoreMenuOpen] = useState(false);
    83→  const [mobileMoreOpen, setMobileMoreOpen] = useState(false);
    84→  const moreRef = useRef<HTMLDivElement>(null);
    85→  const [loadedModule, setLoadedModule] = useState<ModuleKey>(activeModule);
    86→  const avatarRef = useRef<HTMLDivElement>(null);
    87→  const fabRef = useRef<HTMLDivElement>(null);
    88→
    89→  // Skeleton variant for current module
    90→  const skeletonVariant: SkeletonVariant = useMemo(() => getSkeletonVariant(activeModule), [activeModule]);
    91→
    92→  // Show skeleton for 400ms when switching modules
    93→  // loadedModule lags behind activeModule for 400ms to show skeleton
    94→  const isModuleLoading = loadedModule !== activeModule;
    95→  useEffect(() => {
    96→    const timer = setTimeout(() => setLoadedModule(activeModule), 400);
    97→    return () => clearTimeout(timer);
    98→  }, [activeModule]);
    99→
   100→  // Close avatar dropdown on outside click
   101→  useEffect(() => {
   102→    const handler = (e: MouseEvent) => {
   103→      if (avatarRef.current && !avatarRef.current.contains(e.target as Node)) setAvatarOpen(false);
   104→    };
   105→    document.addEventListener('mousedown', handler);
   106→    return () => document.removeEventListener('mousedown', handler);
   107→  }, []);
   108→
   109→  // Close FAB on outside click
   110→  useEffect(() => {
   111→    const handler = (e: MouseEvent) => {
   112→      if (fabRef.current && !fabRef.current.contains(e.target as Node)) setFabOpen(false);
   113→    };
   114→    if (fabOpen) {
   115→      setTimeout(() => document.addEventListener('mousedown', handler), 0);
   116→    }
   117→    return () => document.removeEventListener('mousedown', handler);
   118→  }, [fabOpen]);
   119→
   120→  // Close more menu on outside click
   121→  useEffect(() => {
   122→    const handler = (e: MouseEvent) => {
   123→      if (moreRef.current && !moreRef.current.contains(e.target as Node)) setMoreMenuOpen(false);
   124→    };

   126→    return () => document.removeEventListener('mousedown', handler);
   127→  }, []);

   130→  const handleFullscreen = useCallback(() => {
   131→    if (!document.fullscreenElement) {
   132→      document.documentElement.requestFullscreen().then(() => setIsFullscreen(true)).catch(() => {});
   133→    } else {
   134→      document.exitFullscreen().then(() => setIsFullscreen(false)).catch(() => {});
   135→    }
   136→  }, []);
   137→
   138→  useEffect(() => {
   139→    const handler = () => setIsFullscreen(!!document.fullscreenElement);
   140→    document.addEventListener('fullscreenchange', handler);
   141→    return () => document.removeEventListener('fullscreenchange', handler);
   142→  }, []);
   143→

   145→  useEffect(() => {
   146→    const handler = (e: KeyboardEvent) => {
   147→      // Cmd+K or Ctrl+K for command palette
   148→      if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
   149→        e.preventDefault();
   150→        setCommandPaletteOpen(!commandPaletteOpen);
   151→        return;
   152→      }
   153→      // Number keys 1-9 for module navigation (only when not in input)
   154→      if (!e.metaKey && !e.ctrlKey && !e.altKey) {
   155→        const target = e.target as HTMLElement;
   156→        if (target.tagName === 'INPUT' || target.tagName === 'TEXTAREA' || target.tagName === 'SELECT') return;

   158→        if (num >= 1 && num <= NAV_ITEMS.length) {
   159→          e.preventDefault();
   160→          setActiveModule(NAV_ITEMS[num - 1].key);
   161→        }
   162→        // ? key for help overlay
   163→        if (e.key === '?') {
   164→          e.preventDefault();
   165→          setShowHelpOverlay((p) => !p);
   166→          return;
   167→        }
   168→        // F key for fullscreen toggle
   169→        if (e.key === 'f' || e.key === 'F') {
   170→          e.preventDefault();
   171→          handleFullscreen();
   172→          return;
   173→        }
   174→        if (e.key === 'Escape') {
   175→          setCommandPaletteOpen(false);
   176→          setAvatarOpen(false);
   177→          setFabOpen(false);
   178→          setShowHelpOverlay(false);
   179→        }
   180→      }
   181→    };

   213→      case 'reportes': return <ModuleReportes />;
   214→      case 'incidencias': return <ModuleIncidencias />;

   217→      case 'comunicaciones': return <ModuleComunicaciones />;

   226→    { icon: Zap, label: 'Despacho', color: '#8B5CF6', action: () => { setActiveModule('despacho'); setFabOpen(false); } },
   227→    { icon: BarChart3, label: 'Finanzas', color: '#16A34A', action: () => { setActiveModule('finanzas'); setFabOpen(false); } },

   229→  ];

   232→    <div style={{ display: 'flex', flexDirection: 'column', height: '100vh', overflow: 'hidden', background: 'var(--lf-bg-base)' }}>

   235→        height: 56, flexShrink: 0, display: 'flex', alignItems: 'center', justifyContent: 'space-between',
   236→        padding: '0 20px', background: 'var(--lf-surface)', borderBottom: '1px solid var(--lf-border)',
   237→        zIndex: 100,
   238→      }}>
   239→        {/* Left: Logo + Breadcrumb */}
   240→        <div style={{ display: 'flex', alignItems: 'center', gap: 12 }}>
   241→          <div style={{ display: 'flex', alignItems: 'center', gap: 10 }}>
   242→            <div style={{
   243→              width: 36, height: 36, borderRadius: 8, background: 'var(--lf-accent)',
   244→              display: 'flex', alignItems: 'center', justifyContent: 'center',
   245→              fontFamily: "'DM Mono', monospace", fontWeight: 700, fontSize: 14, color: '#fff',
   246→            }}>LF</div>
   247→            <span className="font-serif" style={{ fontSize: 20, color: 'var(--lf-text-main)', letterSpacing: '-0.02em' }}>LOGIFAST</span>

   250→          {/* Breadcrumb */}

   252→            <span style={{ fontSize: 12, color: 'var(--lf-text-muted)' }}>Dashboard</span>

   328→                      <span style={{ marginLeft: 'auto', fontSize: 11, color: 'var(--lf-text-muted)', fontFamily: "'DM Mono', monospace" }}>{item.shortcut}</span>
   329→                    </button>

   331→                })}
   332→              </motion.div>

   334→          </div>
   335→        </nav>
   336→
   337→        {/* Right: Actions */}
   338→        <div style={{ display: 'flex', alignItems: 'center', gap: 6 }}>
   339→          {/* En vivo indicator */}
   340→          <div

   395→              background: 'var(--lf-surface)', cursor: 'pointer', display: 'flex',
   396→              alignItems: 'center', justifyContent: 'center', color: 'var(--lf-text-muted)',
   397→            }}
   398→            aria-label="Cambiar tema"

   400→            {isDark ? <Sun size={15} /> : <Moon size={15} />}
   401→          </button>
   402→
   403→          {/* Avatar */}

   406→              onClick={() => setAvatarOpen((p) => !p)}
   407→              style={{
   408→                width: 36, height: 36, borderRadius: 8, border: '1px solid var(--lf-border)',

   420→                style={{
   421→                  position: 'absolute', top: 44, right: 0, minWidth: 200,
