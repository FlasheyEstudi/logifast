    29тЖТ    29тЖТfunction isStoreOpen(horario: Record<string, { abre: string; cierra: string }>): { open: boolean; text: string } {
    30тЖТ    30тЖТ  const days = ['dom', 'lun', 'mar', 'mie', 'jue', 'vie', 'sab'];
    31тЖТ    31тЖТ  const now = new Date();
    32тЖТ    32тЖТ  const dayKey = days[now.getDay()];
    33тЖТ    33тЖТ  const dayHorario = horario[dayKey];
    34тЖТ    34тЖТ  if (!dayHorario || !dayHorario.abre || !dayHorario.cierra) {
    35тЖТ    35тЖТ    return { open: false, text: 'Cerrado hoy' };
    36тЖТ    36тЖТ  }
    37тЖТ    37тЖТ  const currentMinutes = now.getHours() * 60 + now.getMinutes();
    38тЖТ    38тЖТ  const [abreH, abreM] = dayHorario.abre.split(':').map(Number);
    39тЖТ    39тЖТ  const [cierraH, cierraM] = dayHorario.cierra.split(':').map(Number);
    40тЖТ    40тЖТ  const abreMinutes = abreH * 60 + abreM;
    41тЖТ    41тЖТ  const cierraMinutes = cierraH * 60 + cierraM;
    42тЖТ    42тЖТ  if (currentMinutes >= abreMinutes && currentMinutes <= cierraMinutes) {
    43тЖТ    43тЖТ    const cierraFormatted = dayHorario.cierra;
    44тЖТ    44тЖТ    return { open: true, text: `Abierto hasta ${cierraFormatted}` };
    45тЖТ    45тЖТ  }
    46тЖТ    46тЖТ  if (currentMinutes < abreMinutes) {
    47тЖТ    47тЖТ    return { open: false, text: `Abre a las ${dayHorario.abre}` };
    48тЖТ    48тЖТ  }
    49тЖТ    49тЖТ  return { open: false, text: 'Cerrado' };
    50тЖТ    50тЖТ}
