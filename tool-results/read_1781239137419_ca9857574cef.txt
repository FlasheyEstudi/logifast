    56тЖТ    56тЖТfunction haversineKm(lat1: number, lng1: number, lat2: number, lng2: number): number {
    57тЖТ    57тЖТ  const R = 6371;
    58тЖТ    58тЖТ  const dLat = ((lat2 - lat1) * Math.PI) / 180;
    59тЖТ    59тЖТ  const dLng = ((lng2 - lng1) * Math.PI) / 180;
    60тЖТ    60тЖТ  const a =
    61тЖТ    61тЖТ    Math.sin(dLat / 2) * Math.sin(dLat / 2) +
    62тЖТ    62тЖТ    Math.cos((lat1 * Math.PI) / 180) *
    63тЖТ    63тЖТ      Math.cos((lat2 * Math.PI) / 180) *
    64тЖТ    64тЖТ      Math.sin(dLng / 2) *
    65тЖТ    65тЖТ      Math.sin(dLng / 2);
    66тЖТ    66тЖТ  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
    67тЖТ    67тЖТ  return R * c;
    68тЖТ    68тЖТ}

    70тЖТ    70тЖТfunction useCountUp(end: number, duration = 800, start = true) {
    71тЖТ    71тЖТ  const [value, setValue] = useState(0);
    72тЖТ    72тЖТ  const rafRef = useRef<number>(0);
    73тЖТ    73тЖТ
    74тЖТ    74тЖТ  useEffect(() => {
    75тЖТ    75тЖТ    if (!start) return;
    76тЖТ    76тЖТ    let startTime: number | null = null;
    77тЖТ    77тЖТ    const from = value;
    78тЖТ    78тЖТ    const animate = (ts: number) => {
    79тЖТ    79тЖТ      if (!startTime) startTime = ts;
    80тЖТ    80тЖТ      const progress = Math.min((ts - startTime) / duration, 1);
    81тЖТ    81тЖТ      const eased = 1 - Math.pow(1 - progress, 3);
    82тЖТ    82тЖТ      const nextVal = Math.round(from + (end - from) * eased);
    83тЖТ    83тЖТ      setValue(nextVal);
    84тЖТ    84тЖТ      if (progress < 1) rafRef.current = requestAnimationFrame(animate);
    85тЖТ    85тЖТ    };
    86тЖТ    86тЖТ    rafRef.current = requestAnimationFrame(animate);
    87тЖТ    87тЖТ    return () => cancelAnimationFrame(rafRef.current);
    88тЖТ    88тЖТ  }, [end, duration, start]);
    89тЖТ    89тЖТ
    90тЖТ    90тЖТ  return start ? value : 0;
    91тЖТ    91тЖТ}
