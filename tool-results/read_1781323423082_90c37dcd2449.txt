    66тЖТ    66тЖТfunction relativeTime(timestamp: string): string {
    67тЖТ    67тЖТ  const now = Date.now();
    68тЖТ    68тЖТ  const then = new Date(timestamp).getTime();
    69тЖТ    69тЖТ  const diff = now - then;
    70тЖТ    70тЖТ
    71тЖТ    71тЖТ  const minutes = Math.floor(diff / 60000);
    72тЖТ    72тЖТ  if (minutes < 1) return 'Ahora mismo';
    73тЖТ    73тЖТ  if (minutes < 60) return `Hace ${minutes} min`;
    74тЖТ    74тЖТ
    75тЖТ    75тЖТ  const hours = Math.floor(minutes / 60);
    76тЖТ    76тЖТ  if (hours < 24) return `Hace ${hours} hora${hours > 1 ? 's' : ''}`;
    77тЖТ    77тЖТ
    78тЖТ    78тЖТ  const days = Math.floor(hours / 24);
    79тЖТ    79тЖТ  if (days < 30) return `Hace ${days} d├нa${days > 1 ? 's' : ''}`;
    80тЖТ    80тЖТ
    81тЖТ    81тЖТ  const months = Math.floor(days / 30);
    82тЖТ    82тЖТ  return `Hace ${months} mes${months > 1 ? 'es' : ''}`;
    83тЖТ    83тЖТ}
